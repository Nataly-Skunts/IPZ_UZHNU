#include <iostream>
using namespace std;

float Task(float x, float y, float z)
{
	//double res;
	float a = 5.0;
	float b = -5.0;
	float const1 = 1;
	_asm
	{

		FINIT
		FLD const1					// 1, st(0)
		FLD x						// a, st(0) = a, st(1) = 1
		FLD y						// b, st(0) = b, st(1) = a, st(0) = 1
		FCOMI ST(0), ST(1)

		jb cond1					// a > b
		je cond2					// a = b
		ja cond3					// a < b

		cond1 :						// x = b / a + 5
		FDIV ST(0), ST(1)
			FADD a
			FST z
			jmp f

		cond2 :						// x = -5
			FLD b
			FST z
			jmp f

		cond3 :						// x = (a * a - b) / b
			FXCH ST(2)
			FMUL ST(0), ST(1)
			FMUL ST(0), ST(1)
			FSUB ST(0), ST(2)
			FDIV ST(0), ST(2)
			FST z
			jmp f
			f :
	}
}

int main()

	cout << "\nEnter b = ";
	cin >> b;
	cout << Task(a, b, c) << endl;

}
